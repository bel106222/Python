# Задание 2
# Дан текстовый файл. Необходимо создать новый файл и 
# записать в него следующую статистику по исходному файлу:
# ■ Количество символов;
# ■ Количество строк;
# ■ Количество гласных букв;
# ■ Количество согласных букв;
# ■ Количество цифр.
print('ЗАДАНИЕ 2')
count_symbols = 0
count_lines = 0
count_vowels = 0
count_consonants = 0
count_digits = 0
lst = []
file = open('text.txt', 'r')
for line in file:
    lst.append(line.strip())
    count_lines += 1
file.close()

for line in lst:
    len_str = len(line)
    while len_str:
        current_symbol = line[len_str - 1]
        count_symbols += 1
        
        if current_symbol.isalpha():
            if current_symbol.lower() in 'уеёыаоэяию':
                count_vowels += 1
            else:
                count_consonants += 1
        elif current_symbol.isdigit():
            count_digits += 1
            
        len_str -= 1

file2 = open('text2.txt', 'w')
file2.write(f'Статистика по файлу {file.name}:\n')
file2.write(f'Количество символов: {count_symbols}\n')
file2.write(f'Количество строк: {count_lines}\n')
file2.write(f'Количество гласных букв: {count_vowels}\n')
file2.write(f'Количество согласных букв: {count_consonants}\n')
file2.write(f'Количество цифр: {count_digits}')
print(f'Результаты выведены в файл {file2.name}')
file2.close()
