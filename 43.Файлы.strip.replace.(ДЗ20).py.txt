# Задание 6
# Дан текстовый файл. Найти и заменить в нем заданное слово.
# Что искать и на что заменять определяется пользователем.
print('ЗАДАНИЕ 6')
find_string = input('Введите слово для поиска (например, \"Задание\"): ')
replace_string = input('Введите слово для замены (например, \"Упражнение\"): ')
lst = []
count = 0
file = open('text.txt', 'r')
for line in file:
    lst.append(line.strip())
file.close()

file2 = open('text2.txt', 'w')
for line in lst:
    if find_string in line:
        line = line.replace(find_string, replace_string)
        count += 1
    file2.write(line + '\n')
file2.close()
print(f'Произведено {count} замен, результат записан в файл {file2.name}')
